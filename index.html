Código Arduino (referencia Wokwi)
#include <DHT.h>
#include <WiFi.h>
#include <HTTPClient.h>

// Configuración de pines
#define DHTPIN   4
#define DHTTYPE  DHT22
#define LED_PIN  2

// Credenciales WiFi
const char* ssid     = "TU_SSID";
const char* password = "TU_PASSWORD";

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(115200);
  dht.begin();
  pinMode(LED_PIN, OUTPUT);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500); Serial.print(".");
  }
  Serial.println("WiFi conectado: " + WiFi.localIP().toString());
}

void loop() {
  float h = dht.readHumidity();
  float t = dht.readTemperature();
  if (isnan(h) || isnan(t)) {
    Serial.println("Error leyendo DHT22!"); return;
  }
  Serial.printf("T: %.1f°C  H: %.1f%%\n", t, h);
  digitalWrite(LED_PIN, t > 28 ? HIGH : LOW);
  delay(15000);
}

Código Arduino BLE (referencia Wokwi)
#include <BLEDevice.h>
#include <BLEServer.h>
#include <BLEUtils.h>

#define SERVICE_UUID        "0000FFE0-0000-1000-8000-00805F9B34FB"
#define CHARACTERISTIC_UUID "0000FFE1-0000-1000-8000-00805F9B34FB"
#define LED_PIN 2

class MyCallbacks : public BLECharacteristicCallbacks {
  void onWrite(BLECharacteristic *c) {
    String val = c->getValue().c_str();
    if (val == "ON")  digitalWrite(LED_PIN, HIGH);
    if (val == "OFF") digitalWrite(LED_PIN, LOW);
    Serial.println("CMD: " + val);
  }
};

void setup() {
  Serial.begin(115200);
  pinMode(LED_PIN, OUTPUT);
  BLEDevice::init("ESP32_Device");
  BLEServer *server = BLEDevice::createServer();
  BLEService *svc   = server->createService(SERVICE_UUID);
  BLECharacteristic *ch = svc->createCharacteristic(
    CHARACTERISTIC_UUID, BLECharacteristic::PROPERTY_WRITE);
  ch->setCallbacks(new MyCallbacks());
  svc->start();
  BLEDevice::startAdvertising();
  Serial.println("BLE listo. Esperando conexión...");
}

Código Arduino — ThingSpeak (referencia Wokwi)
#include <WiFi.h>
#include <ThingSpeak.h>
#include <DHT.h>

const char* ssid     = "TU_SSID";
const char* password = "TU_PASSWORD";
unsigned long channelID = 1234567;
const char*   apiKey    = "TU_API_KEY";

WiFiClient client;
DHT dht(4, DHT22);

void setup() {
  Serial.begin(115200);
  dht.begin();
  pinMode(2, OUTPUT);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(500);
  ThingSpeak.begin(client);
  Serial.println("ThingSpeak conectado.");
}

void loop() {
  float t = dht.readTemperature();
  float h = dht.readHumidity();
  int   ledState = digitalRead(2);
  
  ThingSpeak.setField(1, t);
  ThingSpeak.setField(2, h);
  ThingSpeak.setField(3, ledState);
  ThingSpeak.writeFields(channelID, apiKey);
  
  Serial.printf("Enviado → T:%.1f H:%.1f LED:%d\n", t, h, ledState);
  delay(15000); // ThingSpeak: mín 15s
}
